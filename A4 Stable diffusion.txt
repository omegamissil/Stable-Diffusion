from diffusers import StableDiffusionPipeline
import torch
import time

# Carregar modelo
pipe = StableDiffusionPipeline.from_pretrained("runwayml/stable-diffusion-v1-5", torch_dtype=torch.float16)
pipe = pipe.to("cuda")


# Gerar imagem
prompt = "elephant, in the jungle, and monkeys"

ti = time.time()
image = pipe(prompt).images[0]
image.save("output.png")
tf = time.time()
tt = tf - ti
print(f"Tempo de processamento para gerar a imagem: {tt:.2f} segundos")

image.show()
